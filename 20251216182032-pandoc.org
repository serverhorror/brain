#+title: pandoc
#+filetags: :kb:

Convert a ~docx~ file to ~md~:

#+begin_src shell
  pandoc \
      --verbose \
      --from docx \
      --to markdown \
      --default-image-extension=png \
      --extract-media=images \
      'document.docx' -o document.md
#+end_src
